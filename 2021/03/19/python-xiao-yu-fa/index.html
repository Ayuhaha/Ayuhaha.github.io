def my_range(x):
    a=0;
    while a<x:
        yield a
        a+=1

for x in my_range(20):
    print(x)

steps = ['Python','Git','Deep','AI']

#自己的enumerate，用i和step填充&#123;&#125;&#123;&#125;，iterable和start为形参，start可以提前设置初值
def my_enumerate(iterable,start=0):
    count = start
    for element in iterable:
        yield count,element
        count+=1

for i,step in my_enumerate(steps,1):
    print("Step&#123;&#125;:&#123;&#125;".format(i,step))

#编写生成器对大数据进行拆分，i初值0，0~len(iterable)拆分为长度是size的list
def chunker(iterable,size):
    for i in range(0,len(iterable),size):
        yield iterable[i:i+size]

for chunk in chunker(range(100),20):
    print(list(chunk))

print(len(range(100)))
